    void on_insert_EMPLOYEE(const long employee_employee_id, const STRING_TYPE& employee_last_name, const STRING_TYPE& employee_first_name, const STRING_TYPE& employee_middle_name, const long employee_job_id, const long employee_manager_id, const date employee_hire_date, const DOUBLE_TYPE employee_salary, const DOUBLE_TYPE employee_commission, const long employee_department_id) {
      BEGIN_TRIGGER(exec_stats,"insert_EMPLOYEE")
      BEGIN_TRIGGER(ivc_stats,"insert_EMPLOYEE")
      {  //++tN;
        TOTAL_EMPLOYEES.addOrDelOnZero(se1.modify(employee_department_id),1L);
        TOTAL_SALARY.addOrDelOnZero(se2.modify(employee_department_id),employee_salary);
        DOUBLE_TYPE agg1 = 0.0;
        long l1 = AVG_SALARY_mEMPLOYEE1_L2_1.getValueOrDefault(se4.modify(employee_department_id));
        agg1 += Udiv(l1);
        DOUBLE_TYPE agg2 = 0.0;
        long l2 = (AVG_SALARY_mEMPLOYEE1_L2_1.getValueOrDefault(se6.modify(employee_department_id)) + 1L);
        agg2 += Udiv(l2);
        AVG_SALARY.addOrDelOnZero(se3.modify(employee_department_id),((agg1 * (AVG_SALARY_mEMPLOYEE4.getValueOrDefault(se5.modify(employee_department_id)) * -1L)) + (agg2 * (AVG_SALARY_mEMPLOYEE4.getValueOrDefault(se7.modify(employee_department_id)) + employee_salary))));
        AVG_SALARY_mEMPLOYEE1_L2_1.addOrDelOnZero(se8.modify(employee_department_id),1L);
        AVG_SALARY_mEMPLOYEE4.addOrDelOnZero(se9.modify(employee_department_id),employee_salary);
      }
      END_TRIGGER(exec_stats,"insert_EMPLOYEE")
      END_TRIGGER(ivc_stats,"insert_EMPLOYEE")
    }

    }
    
    void on_delete_EMPLOYEE(const long employee_employee_id, const STRING_TYPE& employee_last_name, const STRING_TYPE& employee_first_name, const STRING_TYPE& employee_middle_name, const long employee_job_id, const long employee_manager_id, const date employee_hire_date, const DOUBLE_TYPE employee_salary, const DOUBLE_TYPE employee_commission, const long employee_department_id) {
      BEGIN_TRIGGER(exec_stats,"delete_EMPLOYEE")
      BEGIN_TRIGGER(ivc_stats,"delete_EMPLOYEE")
      {  //++tN;
        TOTAL_EMPLOYEES.addOrDelOnZero(se10.modify(employee_department_id),-1L);
        TOTAL_SALARY.addOrDelOnZero(se11.modify(employee_department_id),(-1L * employee_salary));
        DOUBLE_TYPE agg3 = 0.0;
        long l3 = (AVG_SALARY_mEMPLOYEE1_L2_1.getValueOrDefault(se13.modify(employee_department_id)) + ((employee_department_id == employee_department_id) * -1L));
        agg3 += Udiv(l3);
        AVG_SALARY.addOrDelOnZero(se12.modify(employee_department_id),((agg3 * employee_salary) * -1L));
        { //foreach
          AVG_SALARY_mEMPLOYEE4_entry* e1 = AVG_SALARY_mEMPLOYEE4.head;
          while(e1){
            long employee_department_id_1 = e1->EMPLOYEE_DEPARTMENT_ID;
            DOUBLE_TYPE v1 = e1->__av;
            DOUBLE_TYPE agg4 = 0.0;
            long l4 = AVG_SALARY_mEMPLOYEE1_L2_1.getValueOrDefault(se14.modify(employee_department_id_1));
            agg4 += Udiv(l4);
            AVG_SALARY.addOrDelOnZero(se12.modify(employee_department_id_1),((v1 * agg4) * -1L));
            e1 = e1->nxt;
          }
        }
        { //foreach
          AVG_SALARY_mEMPLOYEE4_entry* e2 = AVG_SALARY_mEMPLOYEE4.head;
          while(e2){
            long employee_department_id_1 = e2->EMPLOYEE_DEPARTMENT_ID;
            DOUBLE_TYPE v2 = e2->__av;
            DOUBLE_TYPE agg5 = 0.0;
            long l5 = (AVG_SALARY_mEMPLOYEE1_L2_1.getValueOrDefault(se15.modify(employee_department_id_1)) + ((employee_department_id_1 == employee_department_id) * -1L));
            agg5 += Udiv(l5);
            AVG_SALARY.addOrDelOnZero(se12.modify(employee_department_id_1),(v2 * agg5));
            e2 = e2->nxt;
          }
        }
        AVG_SALARY_mEMPLOYEE1_L2_1.addOrDelOnZero(se16.modify(employee_department_id),-1L);
        AVG_SALARY_mEMPLOYEE4.addOrDelOnZero(se17.modify(employee_department_id),(-1L * employee_salary));
      }
      END_TRIGGER(exec_stats,"delete_EMPLOYEE")
      END_TRIGGER(ivc_stats,"delete_EMPLOYEE")

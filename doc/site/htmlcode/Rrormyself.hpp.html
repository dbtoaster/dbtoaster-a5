  /* Definitions of auxiliary maps for storing materialized views. */
  struct COUNT_entry {
    long R1_A; long __av;  COUNT_entry* nxt; COUNT_entry* prv;
    explicit COUNT_entry() : nxt(nullptr), prv(nullptr) { /*R1_A = 0L; __av = 0L; */ }
    explicit COUNT_entry(const long c0, const long c1) { R1_A = c0; __av = c1; }
    COUNT_entry(const COUNT_entry& other) : R1_A( other.R1_A ), __av( other.__av ), nxt( nullptr ), prv( nullptr ) {}
    FORCE_INLINE COUNT_entry& modify(const long c0) { R1_A = c0;  return *this; }
    template<class Archive>
    void serialize(Archive& ar, const unsigned int version) const 
    {
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, R1_A);
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, __av);
    }
  };
  struct COUNT_mapkey0_idxfn {
    FORCE_INLINE static size_t hash(const COUNT_entry& e) {
      size_t h = 0;
      hash_combine(h, e.R1_A);
      return h;
    }
    FORCE_INLINE static bool equals(const COUNT_entry& x, const COUNT_entry& y) {
      return x.R1_A == y.R1_A;
    }
  };
  
  typedef MultiHashMap<COUNT_entry,long,
    HashIndex<COUNT_entry,long,COUNT_mapkey0_idxfn,true>
  > COUNT_map;
  typedef HashIndex<COUNT_entry,long,COUNT_mapkey0_idxfn,true> HashIndex_COUNT_map_0;
  
  struct COUNT_mR1_entry {
    long R2_A; long COUNT_mRR_B; long __av;  COUNT_mR1_entry* nxt; COUNT_mR1_entry* prv;
    explicit COUNT_mR1_entry() : nxt(nullptr), prv(nullptr) { /*R2_A = 0L; COUNT_mRR_B = 0L; __av = 0L; */ }
    explicit COUNT_mR1_entry(const long c0, const long c1, const long c2) { R2_A = c0; COUNT_mRR_B = c1; __av = c2; }
    COUNT_mR1_entry(const COUNT_mR1_entry& other) : R2_A( other.R2_A ), COUNT_mRR_B( other.COUNT_mRR_B ), __av( other.__av ), nxt( nullptr ), prv( nullptr ) {}
    FORCE_INLINE COUNT_mR1_entry& modify(const long c0, const long c1) { R2_A = c0; COUNT_mRR_B = c1;  return *this; }
    FORCE_INLINE COUNT_mR1_entry& modify1(const long c1) { COUNT_mRR_B = c1;  return *this; }
    template<class Archive>
    void serialize(Archive& ar, const unsigned int version) const 
    {
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, R2_A);
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, COUNT_mRR_B);
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, __av);
    }
  };
  struct COUNT_mR1_mapkey01_idxfn {
    FORCE_INLINE static size_t hash(const COUNT_mR1_entry& e) {
      size_t h = 0;
      hash_combine(h, e.R2_A);
      hash_combine(h, e.COUNT_mRR_B);
      return h;
    }
    FORCE_INLINE static bool equals(const COUNT_mR1_entry& x, const COUNT_mR1_entry& y) {
      return x.R2_A == y.R2_A && x.COUNT_mRR_B == y.COUNT_mRR_B;

  struct tlq_t{
    struct timeval t0,t; long tT,tN,tS;

    template<class Archive>
    void serialize(Archive& ar, const unsigned int version) const {
  
      ar << "\n";
      const COUNT_map& _COUNT = get_COUNT();
      dbtoaster::serialize_nvp_tabbed(ar, STRING(COUNT), _COUNT, "\t");
  
    }

      return COUNT;
    
    }

    void on_insert_R(const long r_a, const long r_b) {
      BEGIN_TRIGGER(exec_stats,"insert_R")
      BEGIN_TRIGGER(ivc_stats,"insert_R")
      {  //++tN;
        long agg1 = 0L;
        { //slice
          const HASH_RES_t h1 = COUNT_mR1_mapkey1_idxfn::hash(se2.modify1(r_b));
          const HashIndex_COUNT_mR1_map_1* i1 = static_cast<HashIndex_COUNT_mR1_map_1*>(COUNT_mR1.index[1]);
          HashIndex_COUNT_mR1_map_1::IdxNode* n1 = &(i1->buckets_[h1 % i1->size_]);
          COUNT_mR1_entry* e1;
          do if ((e1=n1->obj) && h1 == n1->hash && COUNT_mR1_mapkey1_idxfn::equals(se2, *e1)) {
            long r2_a = e1->R2_A;
            long v1 = e1->__av;
            long l1 = ((r_a == r2_a) * 2L);
            agg1 += (v1 * (l1 > 0L));
          } while ((n1=n1->next));
        }
        long agg2 = 0L;
        long l2 = ((r_a == r_a) * 2L);
        agg2 += (l2 > 0L);
        COUNT.addOrDelOnZero(se1.modify(r_a),(agg1 + agg2));
        { //slice
          const HASH_RES_t h2 = COUNT_mR1_mapkey1_idxfn::hash(se3.modify1(r_b));
          const HashIndex_COUNT_mR1_map_1* i2 = static_cast<HashIndex_COUNT_mR1_map_1*>(COUNT_mR1.index[1]);
          HashIndex_COUNT_mR1_map_1::IdxNode* n2 = &(i2->buckets_[h2 % i2->size_]);
          COUNT_mR1_entry* e2;
          do if ((e2=n2->obj) && h2 == n2->hash && COUNT_mR1_mapkey1_idxfn::equals(se3, *e2)) {
            long r1_a = e2->R2_A;
            long v2 = e2->__av;
            long agg3 = 0L;
            long l3 = ((r1_a == r_a) * 2L);
            agg3 += (l3 > 0L);
            COUNT.addOrDelOnZero(se1.modify(r1_a),(v2 * agg3));
          } while ((n2=n2->next));
        }
        COUNT_mR1.addOrDelOnZero(se4.modify(r_a,r_b),1L);
      }
      END_TRIGGER(exec_stats,"insert_R")
      END_TRIGGER(ivc_stats,"insert_R")
    }
    void unwrap_insert_R(const event_args_t& ea) {
      on_insert_R(*(reinterpret_cast<long*>(ea[0])), *(reinterpret_cast<long*>(ea[1])));
    }
    
    void on_delete_R(const long r_a, const long r_b) {
      BEGIN_TRIGGER(exec_stats,"delete_R")
      BEGIN_TRIGGER(ivc_stats,"delete_R")
      {  //++tN;
        long agg4 = 0L;
        { //slice
          const HASH_RES_t h3 = COUNT_mR1_mapkey1_idxfn::hash(se6.modify1(r_b));
          const HashIndex_COUNT_mR1_map_1* i3 = static_cast<HashIndex_COUNT_mR1_map_1*>(COUNT_mR1.index[1]);
          HashIndex_COUNT_mR1_map_1::IdxNode* n3 = &(i3->buckets_[h3 % i3->size_]);
          COUNT_mR1_entry* e3;
          do if ((e3=n3->obj) && h3 == n3->hash && COUNT_mR1_mapkey1_idxfn::equals(se6, *e3)) {
            long r2_a = e3->R2_A;
            long v3 = e3->__av;
            long l4 = ((r_a == r2_a) * 2L);
            agg4 += (v3 * (l4 > 0L));
          } while ((n3=n3->next));
        }
        COUNT.addOrDelOnZero(se5.modify(r_a),(agg4 * -1L));
        { //slice
          const HASH_RES_t h4 = COUNT_mR1_mapkey1_idxfn::hash(se7.modify1(r_b));
          const HashIndex_COUNT_mR1_map_1* i4 = static_cast<HashIndex_COUNT_mR1_map_1*>(COUNT_mR1.index[1]);
          HashIndex_COUNT_mR1_map_1::IdxNode* n4 = &(i4->buckets_[h4 % i4->size_]);

            agg5 += (l5 > 0L);
            COUNT.addOrDelOnZero(se5.modify(r1_a),((v4 * agg5) * -1L));
          } while ((n4=n4->next));
        }
        long agg6 = 0L;
        long l6 = ((r_a == r_a) * 2L);
        agg6 += (l6 > 0L);
        COUNT.addOrDelOnZero(se5.modify(r_a),agg6);
        COUNT_mR1.addOrDelOnZero(se8.modify(r_a,r_b),-1L);
      }
      END_TRIGGER(exec_stats,"delete_R")
      END_TRIGGER(ivc_stats,"delete_R")
    }

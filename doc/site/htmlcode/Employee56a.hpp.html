      BEGIN_TRIGGER(exec_stats,"insert_EMPLOYEE")
      BEGIN_TRIGGER(ivc_stats,"insert_EMPLOYEE")
      {  //++tN;
        agg1.clear();
        
        { //foreach
          COUNT_mEMPLOYEE1_entry* e1 = COUNT_mEMPLOYEE1.head;
          while(e1){
            long s_grade_id = e1->S_GRADE_ID;
            DOUBLE_TYPE s_lower_bound = e1->S_LOWER_BOUND;
            DOUBLE_TYPE s_upper_bound = e1->S_UPPER_BOUND;
            long v1 = e1->__av;
            add_to_temp_map(agg1, st1.modify(s_grade_id,(v1 * ((employee_salary >= s_lower_bound) * (s_upper_bound >= employee_salary)))));
            e1 = e1->nxt;

        { //temp foreach
          tuple2_L_L* e2 = agg1.head;

            COUNT.addOrDelOnZero(se1.modify(employee_employee_id,employee_last_name,s_grade_id),v2);
            e2 = e2->nxt;
          }
        }
        COUNT_mSALARY_GRADE1.addOrDelOnZero(se2.modify(employee_employee_id,employee_last_name,employee_salary),1L);
      }
      END_TRIGGER(exec_stats,"insert_EMPLOYEE")
      END_TRIGGER(ivc_stats,"insert_EMPLOYEE")
    }

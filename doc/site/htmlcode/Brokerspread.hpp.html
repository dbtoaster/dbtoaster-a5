  struct BSP_entry {
    long X_BROKER_ID; DOUBLE_TYPE __av;  BSP_entry* nxt; BSP_entry* prv;
    explicit BSP_entry() : nxt(nullptr), prv(nullptr) { /*X_BROKER_ID = 0L; __av = 0.0; */ }
    explicit BSP_entry(const long c0, const DOUBLE_TYPE c1) { X_BROKER_ID = c0; __av = c1; }
    BSP_entry(const BSP_entry& other) : X_BROKER_ID( other.X_BROKER_ID ), __av( other.__av ), nxt( nullptr ), prv( nullptr ) {}
    FORCE_INLINE BSP_entry& modify(const long c0) { X_BROKER_ID = c0;  return *this; }
    template<class Archive>
    void serialize(Archive& ar, const unsigned int version) const 
    {
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, X_BROKER_ID);
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, __av);
    }
  };

  struct tlq_t{
    struct timeval t0,t; long tT,tN,tS;
    tlq_t(): tN(0), tS(0) { gettimeofday(&t0,NULL); }
  
  /* Serialization Code */
    template<class Archive>
    void serialize(Archive& ar, const unsigned int version) const {
  
      ar << "\n";
      const BSP_map& _BSP = get_BSP();
      dbtoaster::serialize_nvp_tabbed(ar, STRING(BSP), _BSP, "\t");
  
    }
  
    /* Functions returning / computing the results of top level queries */
    const BSP_map& get_BSP() const {
      return BSP;
    
    }
  
  protected:
  
    /* Data structures used for storing / computing top level queries */
    BSP_map BSP;
  
  };

    /* Registering relations and trigger functions */
    void register_data(ProgramBase& pb) {
  
      pb.add_map<BSP_map>( "BSP", BSP );
      pb.add_map<BSP_mBIDS1_map>( "BSP_mBIDS1", BSP_mBIDS1 );
      pb.add_map<BSP_mBIDS5_map>( "BSP_mBIDS5", BSP_mBIDS5 );
  
      pb.add_relation("BIDS");
  
      
  
      pb.add_trigger("BIDS", insert_tuple, std::bind(&data_t::unwrap_insert_BIDS, this, std::placeholders::_1));
      pb.add_trigger("BIDS", delete_tuple, std::bind(&data_t::unwrap_delete_BIDS, this, std::placeholders::_1));
  
      #ifdef DBT_PROFILE
      exec_stats = pb.exec_stats;
      ivc_stats = pb.ivc_stats;
      exec_stats->register_probe(0, "insert_BIDS_s0");
      exec_stats->register_probe(1, "insert_BIDS_s1");
      exec_stats->register_probe(2, "insert_BIDS_s2");
      exec_stats->register_probe(3, "delete_BIDS_s0");
      exec_stats->register_probe(4, "delete_BIDS_s1");
      exec_stats->register_probe(5, "delete_BIDS_s2");
      #endif // DBT_PROFILE
  
    }

    /* Trigger functions for stream relations */
    void on_insert_BIDS(const DOUBLE_TYPE bids_t, const long bids_id, const long bids_broker_id, const DOUBLE_TYPE bids_volume, const DOUBLE_TYPE bids_price) {
      BEGIN_TRIGGER(exec_stats,"insert_BIDS")
      BEGIN_TRIGGER(ivc_stats,"insert_BIDS")
      {  //++tN;
        long agg1 = 0L;
        { //slice
          const HASH_RES_t h1 = BSP_mBIDS1_mapkey1_idxfn::hash(se2.modify1(bids_broker_id));
          const HashIndex_BSP_mBIDS1_map_1* i1 = static_cast<HashIndex_BSP_mBIDS1_map_1*>(BSP_mBIDS1.index[1]);
          HashIndex_BSP_mBIDS1_map_1::IdxNode* n1 = &(i1->buckets_[h1 % i1->size_]);
          BSP_mBIDS1_entry* e1;
          do if ((e1=n1->obj) && h1 == n1->hash && BSP_mBIDS1_mapkey1_idxfn::equals(se2, *e1)) {
            DOUBLE_TYPE y_t = e1->Y_T;
            long v1 = e1->__av;
            agg1 += (v1 * (bids_t > y_t));
          } while ((n1=n1->next));
        }
        long agg2 = 0L;
        { //slice
          const HASH_RES_t h2 = BSP_mBIDS1_mapkey1_idxfn::hash(se3.modify1(bids_broker_id));
          const HashIndex_BSP_mBIDS1_map_1* i2 = static_cast<HashIndex_BSP_mBIDS1_map_1*>(BSP_mBIDS1.index[1]);
          HashIndex_BSP_mBIDS1_map_1::IdxNode* n2 = &(i2->buckets_[h2 % i2->size_]);
          BSP_mBIDS1_entry* e2;
          do if ((e2=n2->obj) && h2 == n2->hash && BSP_mBIDS1_mapkey1_idxfn::equals(se3, *e2)) {
            DOUBLE_TYPE x_t = e2->Y_T;
            long v2 = e2->__av;
            agg2 += (v2 * (x_t > bids_t));
          } while ((n2=n2->next));
        }
        DOUBLE_TYPE agg3 = 0.0;
        { //slice
          const HASH_RES_t h3 = BSP_mBIDS5_mapkey1_idxfn::hash(se4.modify1(bids_broker_id));
          const HashIndex_BSP_mBIDS5_map_1* i3 = static_cast<HashIndex_BSP_mBIDS5_map_1*>(BSP_mBIDS5.index[1]);
          HashIndex_BSP_mBIDS5_map_1::IdxNode* n3 = &(i3->buckets_[h3 % i3->size_]);
          BSP_mBIDS5_entry* e3;
          do if ((e3=n3->obj) && h3 == n3->hash && BSP_mBIDS5_mapkey1_idxfn::equals(se4, *e3)) {
            DOUBLE_TYPE x_t = e3->X_T;
            DOUBLE_TYPE v3 = e3->__av;
            agg3 += (v3 * (x_t > bids_t));
          } while ((n3=n3->next));
        }
        DOUBLE_TYPE agg4 = 0.0;
        { //slice
          const HASH_RES_t h4 = BSP_mBIDS5_mapkey1_idxfn::hash(se5.modify1(bids_broker_id));
          const HashIndex_BSP_mBIDS5_map_1* i4 = static_cast<HashIndex_BSP_mBIDS5_map_1*>(BSP_mBIDS5.index[1]);
          HashIndex_BSP_mBIDS5_map_1::IdxNode* n4 = &(i4->buckets_[h4 % i4->size_]);
          BSP_mBIDS5_entry* e4;
          do if ((e4=n4->obj) && h4 == n4->hash && BSP_mBIDS5_mapkey1_idxfn::equals(se5, *e4)) {
            DOUBLE_TYPE y_t = e4->X_T;
            DOUBLE_TYPE v4 = e4->__av;
            agg4 += (v4 * (bids_t > y_t));
          } while ((n4=n4->next));
        }
        BSP.addOrDelOnZero(se1.modify(bids_broker_id),(((agg1 + (agg2 * -1L)) * (bids_volume * bids_price)) + (agg3 + (agg4 * -1L))));
        BSP_mBIDS1.addOrDelOnZero(se6.modify(bids_t,bids_broker_id),1L);
        BSP_mBIDS5.addOrDelOnZero(se7.modify(bids_t,bids_broker_id),(bids_volume * bids_price));
      }
      END_TRIGGER(exec_stats,"insert_BIDS")
      END_TRIGGER(ivc_stats,"insert_BIDS")
    }
    void unwrap_insert_BIDS(const event_args_t& ea) {
      on_insert_BIDS(*(reinterpret_cast<DOUBLE_TYPE*>(ea[0])), *(reinterpret_cast<long*>(ea[1])), *(reinterpret_cast<long*>(ea[2])), *(reinterpret_cast<DOUBLE_TYPE*>(ea[3])), *(reinterpret_cast<DOUBLE_TYPE*>(ea[4])));
    }

            DOUBLE_TYPE y_t = e7->X_T;
            DOUBLE_TYPE v7 = e7->__av;
            agg7 += (v7 * (bids_t > y_t));
          } while ((n7=n7->next));
        }
        long agg8 = 0L;
        { //slice
          const HASH_RES_t h8 = BSP_mBIDS1_mapkey1_idxfn::hash(se12.modify1(bids_broker_id));
          const HashIndex_BSP_mBIDS1_map_1* i8 = static_cast<HashIndex_BSP_mBIDS1_map_1*>(BSP_mBIDS1.index[1]);
          HashIndex_BSP_mBIDS1_map_1::IdxNode* n8 = &(i8->buckets_[h8 % i8->size_]);
          BSP_mBIDS1_entry* e8;
          do if ((e8=n8->obj) && h8 == n8->hash && BSP_mBIDS1_mapkey1_idxfn::equals(se12, *e8)) {
            DOUBLE_TYPE x_t = e8->Y_T;
            long v8 = e8->__av;
            agg8 += (v8 * (x_t > bids_t));
          } while ((n8=n8->next));
        }
        BSP.addOrDelOnZero(se8.modify(bids_broker_id),((((agg5 * (bids_volume * bids_price)) + agg6) * -1L) + (agg7 + (agg8 * (bids_volume * bids_price)))));
        BSP_mBIDS1.addOrDelOnZero(se13.modify(bids_t,bids_broker_id),-1L);
        BSP_mBIDS5.addOrDelOnZero(se14.modify(bids_t,bids_broker_id),(-1L * (bids_volume * bids_price)));
      }
      END_TRIGGER(exec_stats,"delete_BIDS")
      END_TRIGGER(ivc_stats,"delete_BIDS")

  /* Definitions of auxiliary maps for storing materialized views. */
  struct BSV_entry {
    long X_BROKER_ID; DOUBLE_TYPE __av;  BSV_entry* nxt; BSV_entry* prv;
    explicit BSV_entry() : nxt(nullptr), prv(nullptr) { /*X_BROKER_ID = 0L; __av = 0.0; */ }
    explicit BSV_entry(const long c0, const DOUBLE_TYPE c1) { X_BROKER_ID = c0; __av = c1; }
    BSV_entry(const BSV_entry& other) : X_BROKER_ID( other.X_BROKER_ID ), __av( other.__av ), nxt( nullptr ), prv( nullptr ) {}
    FORCE_INLINE BSV_entry& modify(const long c0) { X_BROKER_ID = c0;  return *this; }
    template<class Archive>
    void serialize(Archive& ar, const unsigned int version) const 
    {
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, X_BROKER_ID);
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, __av);
    }
  };
  struct BSV_mapkey0_idxfn {
    FORCE_INLINE static size_t hash(const BSV_entry& e) {
      size_t h = 0;
      hash_combine(h, e.X_BROKER_ID);
      return h;
    }
    FORCE_INLINE static bool equals(const BSV_entry& x, const BSV_entry& y) {
      return x.X_BROKER_ID == y.X_BROKER_ID;
    }
  };
  
  typedef MultiHashMap<BSV_entry,DOUBLE_TYPE,
    HashIndex<BSV_entry,DOUBLE_TYPE,BSV_mapkey0_idxfn,true>
  > BSV_map;
  typedef HashIndex<BSV_entry,DOUBLE_TYPE,BSV_mapkey0_idxfn,true> HashIndex_BSV_map_0;
  
  struct BSV_mBIDS1_entry {
    long BSV_mBIDSBIDS_BROKER_ID; DOUBLE_TYPE __av;  BSV_mBIDS1_entry* nxt; BSV_mBIDS1_entry* prv;
    explicit BSV_mBIDS1_entry() : nxt(nullptr), prv(nullptr) { /*BSV_mBIDSBIDS_BROKER_ID = 0L; __av = 0.0; */ }
    explicit BSV_mBIDS1_entry(const long c0, const DOUBLE_TYPE c1) { BSV_mBIDSBIDS_BROKER_ID = c0; __av = c1; }
    BSV_mBIDS1_entry(const BSV_mBIDS1_entry& other) : BSV_mBIDSBIDS_BROKER_ID( other.BSV_mBIDSBIDS_BROKER_ID ), __av( other.__av ), nxt( nullptr ), prv( nullptr ) {}
    FORCE_INLINE BSV_mBIDS1_entry& modify(const long c0) { BSV_mBIDSBIDS_BROKER_ID = c0;  return *this; }
    template<class Archive>
    void serialize(Archive& ar, const unsigned int version) const 
    {
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, BSV_mBIDSBIDS_BROKER_ID);
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, __av);
    }
  };

    }
    FORCE_INLINE static bool equals(const BSV_mBIDS1_entry& x, const BSV_mBIDS1_entry& y) {
      return x.BSV_mBIDSBIDS_BROKER_ID == y.BSV_mBIDSBIDS_BROKER_ID;
    }
  };

  /* Type definition providing a way to access the results of the sql program */
  struct tlq_t{
    struct timeval t0,t; long tT,tN,tS;
    tlq_t(): tN(0), tS(0) { gettimeofday(&t0,NULL); }
  
  /* Serialization Code */
    template<class Archive>
    void serialize(Archive& ar, const unsigned int version) const {
  
      ar << "\n";
      const BSV_map& _BSV = get_BSV();
      dbtoaster::serialize_nvp_tabbed(ar, STRING(BSV), _BSV, "\t");
  
    }
  
    /* Functions returning / computing the results of top level queries */
    const BSV_map& get_BSV() const {
      return BSV;
    
    }
  
  protected:
  
    /* Data structures used for storing / computing top level queries */
    BSV_map BSV;
  
  };
  
  /* Type definition providing a way to incrementally maintain the results of the sql program */
  struct data_t : tlq_t{
    data_t(): tlq_t() {
      
    }
  
    #ifdef DBT_PROFILE
    std::shared_ptr<dbtoaster::statistics::trigger_exec_stats> exec_stats;
    std::shared_ptr<dbtoaster::statistics::trigger_exec_stats> ivc_stats;
    #endif
  
    /* Registering relations and trigger functions */
    void register_data(ProgramBase& pb) {
  
      pb.add_map<BSV_map>( "BSV", BSV );
      pb.add_map<BSV_mBIDS1_map>( "BSV_mBIDS1", BSV_mBIDS1 );
  
      pb.add_relation("BIDS");
  
      
  
      pb.add_trigger("BIDS", insert_tuple, std::bind(&data_t::unwrap_insert_BIDS, this, std::placeholders::_1));
      pb.add_trigger("BIDS", delete_tuple, std::bind(&data_t::unwrap_delete_BIDS, this, std::placeholders::_1));
  
      #ifdef DBT_PROFILE
      exec_stats = pb.exec_stats;
      ivc_stats = pb.ivc_stats;
      exec_stats->register_probe(0, "insert_BIDS_s0");
      exec_stats->register_probe(1, "insert_BIDS_s1");
      exec_stats->register_probe(2, "delete_BIDS_s0");
      exec_stats->register_probe(3, "delete_BIDS_s1");
      #endif // DBT_PROFILE
  
    }
  
    /* Trigger functions for table relations */
    
    
    /* Trigger functions for stream relations */
    void on_insert_BIDS(const DOUBLE_TYPE bids_t, const long bids_id, const long bids_broker_id, const DOUBLE_TYPE bids_volume, const DOUBLE_TYPE bids_price) {
      BEGIN_TRIGGER(exec_stats,"insert_BIDS")
      BEGIN_TRIGGER(ivc_stats,"insert_BIDS")
      {  //++tN;
        BSV.addOrDelOnZero(se1.modify(bids_broker_id),(((BSV_mBIDS1.getValueOrDefault(se2.modify(bids_broker_id)) * (2L * (bids_price * bids_volume))) + ((bids_price * bids_price) * (bids_volume * bids_volume))) * 0.5));
        BSV_mBIDS1.addOrDelOnZero(se3.modify(bids_broker_id),(bids_price * bids_volume));
      }
      END_TRIGGER(exec_stats,"insert_BIDS")
      END_TRIGGER(ivc_stats,"insert_BIDS")
    }
    void unwrap_insert_BIDS(const event_args_t& ea) {
      on_insert_BIDS(*(reinterpret_cast<DOUBLE_TYPE*>(ea[0])), *(reinterpret_cast<long*>(ea[1])), *(reinterpret_cast<long*>(ea[2])), *(reinterpret_cast<DOUBLE_TYPE*>(ea[3])), *(reinterpret_cast<DOUBLE_TYPE*>(ea[4])));
    }
    
    void on_delete_BIDS(const DOUBLE_TYPE bids_t, const long bids_id, const long bids_broker_id, const DOUBLE_TYPE bids_volume, const DOUBLE_TYPE bids_price) {
      BEGIN_TRIGGER(exec_stats,"delete_BIDS")
      BEGIN_TRIGGER(ivc_stats,"delete_BIDS")
      {  //++tN;
        BSV.addOrDelOnZero(se4.modify(bids_broker_id),((BSV_mBIDS1.getValueOrDefault(se5.modify(bids_broker_id)) * (-1.0 * (bids_price * bids_volume))) + (0.5 * ((bids_price * bids_price) * (bids_volume * bids_volume)))));
        BSV_mBIDS1.addOrDelOnZero(se6.modify(bids_broker_id),(-1L * (bids_price * bids_volume)));
      }
      END_TRIGGER(exec_stats,"delete_BIDS")
      END_TRIGGER(ivc_stats,"delete_BIDS")
    }
    void unwrap_delete_BIDS(const event_args_t& ea) {
      on_delete_BIDS(*(reinterpret_cast<DOUBLE_TYPE*>(ea[0])), *(reinterpret_cast<long*>(ea[1])), *(reinterpret_cast<long*>(ea[2])), *(reinterpret_cast<DOUBLE_TYPE*>(ea[3])), *(reinterpret_cast<DOUBLE_TYPE*>(ea[4])));
    }
    
    void on_system_ready_event() {
      BEGIN_TRIGGER(exec_stats,"system_ready_event")
      BEGIN_TRIGGER(ivc_stats,"system_ready_event")
      {  //
        
      }
      END_TRIGGER(exec_stats,"system_ready_event")
      END_TRIGGER(ivc_stats,"system_ready_event")
    }
  
  private:
  
    /* Sample entries for avoiding recreation of temporary objects */
    BSV_entry se1;
    BSV_mBIDS1_entry se2;
    BSV_mBIDS1_entry se3;
    BSV_entry se4;
    BSV_mBIDS1_entry se5;
    BSV_mBIDS1_entry se6;
  
    /* Data structures used for storing materialized views */
    BSV_mBIDS1_map BSV_mBIDS1;
    
    
  
  };

  struct data_t : tlq_t{
    data_t(): tlq_t() {
      
    }
  
    #ifdef DBT_PROFILE
    std::shared_ptr<dbtoaster::statistics::trigger_exec_stats> exec_stats;
    std::shared_ptr<dbtoaster::statistics::trigger_exec_stats> ivc_stats;
    #endif
  
    /* Registering relations and trigger functions */
    void register_data(ProgramBase& pb) {
  
      pb.add_map<QUERY11A_map>( "QUERY11A", QUERY11A );
      pb.add_map<QUERY11A_mSUPPLIER1_map>( "QUERY11A_mSUPPLIER1", QUERY11A_mSUPPLIER1 );
      pb.add_map<QUERY11A_mPARTSUPP1_map>( "QUERY11A_mPARTSUPP1", QUERY11A_mPARTSUPP1 );
  
      pb.add_relation("SUPPLIER");
      pb.add_relation("PARTSUPP");
  
      
  
      pb.add_trigger("SUPPLIER", insert_tuple, std::bind(&data_t::unwrap_insert_SUPPLIER, this, std::placeholders::_1));
      pb.add_trigger("SUPPLIER", delete_tuple, std::bind(&data_t::unwrap_delete_SUPPLIER, this, std::placeholders::_1));
      pb.add_trigger("PARTSUPP", insert_tuple, std::bind(&data_t::unwrap_insert_PARTSUPP, this, std::placeholders::_1));
      pb.add_trigger("PARTSUPP", delete_tuple, std::bind(&data_t::unwrap_delete_PARTSUPP, this, std::placeholders::_1));
  
      #ifdef DBT_PROFILE
      exec_stats = pb.exec_stats;
      ivc_stats = pb.ivc_stats;
      exec_stats->register_probe(0, "insert_SUPPLIER_s0");
      exec_stats->register_probe(1, "insert_SUPPLIER_s1");
      exec_stats->register_probe(2, "delete_SUPPLIER_s0");
      exec_stats->register_probe(3, "delete_SUPPLIER_s1");
      exec_stats->register_probe(4, "insert_PARTSUPP_s0");
      exec_stats->register_probe(5, "insert_PARTSUPP_s1");
      exec_stats->register_probe(6, "delete_PARTSUPP_s0");
      exec_stats->register_probe(7, "delete_PARTSUPP_s1");
      #endif // DBT_PROFILE
  
    }
  
    /* Trigger functions for table relations */
    
    
    /* Trigger functions for stream relations */
    void on_insert_SUPPLIER(const long supplier_suppkey, const STRING_TYPE& supplier_name, const STRING_TYPE& supplier_address, const long supplier_nationkey, const STRING_TYPE& supplier_phone, const DOUBLE_TYPE supplier_acctbal, const STRING_TYPE& supplier_comment) {
      BEGIN_TRIGGER(exec_stats,"insert_SUPPLIER")
      BEGIN_TRIGGER(ivc_stats,"insert_SUPPLIER")
      {  //++tN;
        { //slice
          const HASH_RES_t h1 = QUERY11A_mSUPPLIER1_mapkey1_idxfn::hash(se2.modify1(supplier_suppkey));
          const HashIndex_QUERY11A_mSUPPLIER1_map_1* i1 = static_cast<HashIndex_QUERY11A_mSUPPLIER1_map_1*>(QUERY11A_mSUPPLIER1.index[1]);
          HashIndex_QUERY11A_mSUPPLIER1_map_1::IdxNode* n1 = &(i1->buckets_[h1 % i1->size_]);
          QUERY11A_mSUPPLIER1_entry* e1;
          do if ((e1=n1->obj) && h1 == n1->hash && QUERY11A_mSUPPLIER1_mapkey1_idxfn::equals(se2, *e1)) {
            long ps_partkey = e1->PS_PARTKEY;
            DOUBLE_TYPE v1 = e1->__av;
            QUERY11A.addOrDelOnZero(se1.modify(ps_partkey),v1);
          } while ((n1=n1->next));
        }
        QUERY11A_mPARTSUPP1.addOrDelOnZero(se3.modify(supplier_suppkey),1L);
      }
      END_TRIGGER(exec_stats,"insert_SUPPLIER")
      END_TRIGGER(ivc_stats,"insert_SUPPLIER")
    }
    void unwrap_insert_SUPPLIER(const event_args_t& ea) {
      on_insert_SUPPLIER(*(reinterpret_cast<long*>(ea[0])), *(reinterpret_cast<STRING_TYPE*>(ea[1])), *(reinterpret_cast<STRING_TYPE*>(ea[2])), *(reinterpret_cast<long*>(ea[3])), *(reinterpret_cast<STRING_TYPE*>(ea[4])), *(reinterpret_cast<DOUBLE_TYPE*>(ea[5])), *(reinterpret_cast<STRING_TYPE*>(ea[6])));
    }
    
    void on_delete_SUPPLIER(const long supplier_suppkey, const STRING_TYPE& supplier_name, const STRING_TYPE& supplier_address, const long supplier_nationkey, const STRING_TYPE& supplier_phone, const DOUBLE_TYPE supplier_acctbal, const STRING_TYPE& supplier_comment) {
      BEGIN_TRIGGER(exec_stats,"delete_SUPPLIER")
      BEGIN_TRIGGER(ivc_stats,"delete_SUPPLIER")
      {  //++tN;
        { //slice
          const HASH_RES_t h2 = QUERY11A_mSUPPLIER1_mapkey1_idxfn::hash(se5.modify1(supplier_suppkey));
          const HashIndex_QUERY11A_mSUPPLIER1_map_1* i2 = static_cast<HashIndex_QUERY11A_mSUPPLIER1_map_1*>(QUERY11A_mSUPPLIER1.index[1]);
          HashIndex_QUERY11A_mSUPPLIER1_map_1::IdxNode* n2 = &(i2->buckets_[h2 % i2->size_]);
          QUERY11A_mSUPPLIER1_entry* e2;
          do if ((e2=n2->obj) && h2 == n2->hash && QUERY11A_mSUPPLIER1_mapkey1_idxfn::equals(se5, *e2)) {
            long ps_partkey = e2->PS_PARTKEY;
            DOUBLE_TYPE v2 = e2->__av;
            QUERY11A.addOrDelOnZero(se4.modify(ps_partkey),(v2 * -1L));
          } while ((n2=n2->next));
        }
        QUERY11A_mPARTSUPP1.addOrDelOnZero(se6.modify(supplier_suppkey),-1L);
      }
      END_TRIGGER(exec_stats,"delete_SUPPLIER")
      END_TRIGGER(ivc_stats,"delete_SUPPLIER")
    }
    void unwrap_delete_SUPPLIER(const event_args_t& ea) {
      on_delete_SUPPLIER(*(reinterpret_cast<long*>(ea[0])), *(reinterpret_cast<STRING_TYPE*>(ea[1])), *(reinterpret_cast<STRING_TYPE*>(ea[2])), *(reinterpret_cast<long*>(ea[3])), *(reinterpret_cast<STRING_TYPE*>(ea[4])), *(reinterpret_cast<DOUBLE_TYPE*>(ea[5])), *(reinterpret_cast<STRING_TYPE*>(ea[6])));
    }
    
    void on_insert_PARTSUPP(const long partsupp_partkey, const long partsupp_suppkey, const long partsupp_availqty, const DOUBLE_TYPE partsupp_supplycost, const STRING_TYPE& partsupp_comment) {
      BEGIN_TRIGGER(exec_stats,"insert_PARTSUPP")

        QUERY11A_mSUPPLIER1.addOrDelOnZero(se9.modify(partsupp_partkey,partsupp_suppkey),(partsupp_supplycost * partsupp_availqty));
      }
      END_TRIGGER(exec_stats,"insert_PARTSUPP")
      END_TRIGGER(ivc_stats,"insert_PARTSUPP")
    }

      on_insert_PARTSUPP(*(reinterpret_cast<long*>(ea[0])), *(reinterpret_cast<long*>(ea[1])), *(reinterpret_cast<long*>(ea[2])), *(reinterpret_cast<DOUBLE_TYPE*>(ea[3])), *(reinterpret_cast<STRING_TYPE*>(ea[4])));
    }

      BEGIN_TRIGGER(exec_stats,"delete_PARTSUPP")
      BEGIN_TRIGGER(ivc_stats,"delete_PARTSUPP")
      {  //++tN;
        QUERY11A.addOrDelOnZero(se10.modify(partsupp_partkey),(QUERY11A_mPARTSUPP1.getValueOrDefault(se11.modify(partsupp_suppkey)) * (-1L * (partsupp_supplycost * partsupp_availqty))));
        QUERY11A_mSUPPLIER1.addOrDelOnZero(se12.modify(partsupp_partkey,partsupp_suppkey),(-1L * (partsupp_supplycost * partsupp_availqty)));
      }
      END_TRIGGER(exec_stats,"delete_PARTSUPP")
      END_TRIGGER(ivc_stats,"delete_PARTSUPP")
    }

      on_delete_PARTSUPP(*(reinterpret_cast<long*>(ea[0])), *(reinterpret_cast<long*>(ea[1])), *(reinterpret_cast<long*>(ea[2])), *(reinterpret_cast<DOUBLE_TYPE*>(ea[3])), *(reinterpret_cast<STRING_TYPE*>(ea[4])));
    }

    }

    QUERY11A_entry se10;
    QUERY11A_mPARTSUPP1_entry se11;
    QUERY11A_mSUPPLIER1_entry se12;
  
    /* Data structures used for storing materialized views */
    QUERY11A_mSUPPLIER1_map QUERY11A_mSUPPLIER1;
    QUERY11A_mPARTSUPP1_map QUERY11A_mPARTSUPP1;
    
    
  
  };

  /* Definitions of auxiliary maps for storing materialized views. */
  struct COUNT_entry {
    long BOB; long JOE; long __av;  COUNT_entry* nxt; COUNT_entry* prv;
    explicit COUNT_entry() : nxt(nullptr), prv(nullptr) { /*BOB = 0L; JOE = 0L; __av = 0L; */ }
    explicit COUNT_entry(const long c0, const long c1, const long c2) { BOB = c0; JOE = c1; __av = c2; }
    COUNT_entry(const COUNT_entry& other) : BOB( other.BOB ), JOE( other.JOE ), __av( other.__av ), nxt( nullptr ), prv( nullptr ) {}
    FORCE_INLINE COUNT_entry& modify(const long c0, const long c1) { BOB = c0; JOE = c1;  return *this; }
    template<class Archive>
    void serialize(Archive& ar, const unsigned int version) const 
    {
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, BOB);
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, JOE);
      ar << ELEM_SEPARATOR;
      DBT_SERIALIZATION_NVP(ar, __av);
    }
  };
  struct COUNT_mapkey01_idxfn {
    FORCE_INLINE static size_t hash(const COUNT_entry& e) {
      size_t h = 0;
      hash_combine(h, e.BOB);
      hash_combine(h, e.JOE);
      return h;
    }
    FORCE_INLINE static bool equals(const COUNT_entry& x, const COUNT_entry& y) {
      return x.BOB == y.BOB && x.JOE == y.JOE;
    }
  };
  
  typedef MultiHashMap<COUNT_entry,long,
    HashIndex<COUNT_entry,long,COUNT_mapkey01_idxfn,true>
  > COUNT_map;
  typedef HashIndex<COUNT_entry,long,COUNT_mapkey01_idxfn,true> HashIndex_COUNT_map_01;

  struct tuple2_L_L {
    long _1; long __av; tuple2_L_L* nxt; tuple2_L_L* prv;
    explicit tuple2_L_L() : nxt(nullptr), prv(nullptr) { }
    explicit tuple2_L_L(const long c1, long c__av=0L) : nxt(nullptr), prv(nullptr) { _1 = c1; __av = c__av;}
    int operator<(const tuple2_L_L &rhs) const { 
      if(this->_1!=rhs._1) return (this->_1<rhs._1);
      return 0;
    }
    int operator==(const tuple2_L_L &rhs) const { return ((this->_1==rhs._1)); }
    FORCE_INLINE tuple2_L_L& modify(const long c0, long c__av) { _1 = c0; __av = c__av; return *this; }
    static bool equals(const tuple2_L_L &x, const tuple2_L_L &y) { return ((x._1==y._1)); }
    static long hash(const tuple2_L_L &e) {
      size_t h = 0;
      hash_combine(h, e._1);
      return h;
    }
  };
  
  /* Type definition providing a way to access the results of the sql program */
  struct tlq_t{
    struct timeval t0,t; long tT,tN,tS;
    tlq_t(): tN(0), tS(0) { gettimeofday(&t0,NULL); }
  
  /* Serialization Code */
    template<class Archive>
    void serialize(Archive& ar, const unsigned int version) const {
  
      ar << "\n";
      const COUNT_map& _COUNT = get_COUNT();
      dbtoaster::serialize_nvp_tabbed(ar, STRING(COUNT), _COUNT, "\t");
  
    }
  
    /* Functions returning / computing the results of top level queries */
    const COUNT_map& get_COUNT() const {
      return COUNT;
    
    }
  
  protected:
  
    /* Data structures used for storing / computing top level queries */
    COUNT_map COUNT;
  
  };
  
  /* Type definition providing a way to incrementally maintain the results of the sql program */
  struct data_t : tlq_t{
    data_t(): tlq_t(), agg2(16U), add2(16U), agg1(16U), add1(16U) {
      
    }
  
    #ifdef DBT_PROFILE
    std::shared_ptr<dbtoaster::statistics::trigger_exec_stats> exec_stats;
    std::shared_ptr<dbtoaster::statistics::trigger_exec_stats> ivc_stats;
    #endif
  
    /* Registering relations and trigger functions */
    void register_data(ProgramBase& pb) {
  
      pb.add_map<COUNT_map>( "COUNT", COUNT );
      pb.add_map<COUNT_mR2_map>( "COUNT_mR2", COUNT_mR2 );
      pb.add_map<COUNT_mR5_map>( "COUNT_mR5", COUNT_mR5 );
  
      pb.add_relation("R");
  
      
  
      pb.add_trigger("R", insert_tuple, std::bind(&data_t::unwrap_insert_R, this, std::placeholders::_1));
      pb.add_trigger("R", delete_tuple, std::bind(&data_t::unwrap_delete_R, this, std::placeholders::_1));
  
      #ifdef DBT_PROFILE
      exec_stats = pb.exec_stats;
      ivc_stats = pb.ivc_stats;
      exec_stats->register_probe(0, "insert_R_s0");
      exec_stats->register_probe(1, "insert_R_s1");
      exec_stats->register_probe(2, "insert_R_s2");
      exec_stats->register_probe(3, "delete_R_s0");
      exec_stats->register_probe(4, "delete_R_s1");
      exec_stats->register_probe(5, "delete_R_s2");
      #endif // DBT_PROFILE
  
    }
  
    /* Trigger functions for table relations */
    
    
    /* Trigger functions for stream relations */
    void on_insert_R(const long r_a, const long r_b) {
      BEGIN_TRIGGER(exec_stats,"insert_R")
      BEGIN_TRIGGER(ivc_stats,"insert_R")
      {  //++tN;
        add1.clear();
        { //foreach
          COUNT_mR2_entry* e1 = COUNT_mR2.head;
          while(e1){
            long r2_a = e1->R2_A;
            long joe = e1->JOE;
            long v2 = e1->__av;
            add_to_temp_map(add1, st1.modify(joe,(v2 * (r_b > r2_a))));
            e1 = e1->nxt;
          }
        }
        add_to_temp_map(add1, st2.modify(r_b,(r_b > r_a)));
        { //temp foreach
          tuple2_L_L* e2 = add1.head;
          while(e2){
            long joe = e2->_1;
            long v3 = e2->__av;
            COUNT.addOrDelOnZero(se1.modify(r_a,joe),v3);
            e2 = e2->nxt;
          }
        }
        agg1.clear();
        
        { //foreach
          COUNT_mR5_entry* e3 = COUNT_mR5.head;
          while(e3){
            long r1_b = e3->R1_B;
            long bob = e3->BOB;
            long v4 = e3->__av;
            add_to_temp_map(agg1, st3.modify(bob,(v4 * (r1_b > r_a))));
            e3 = e3->nxt;
          }
        }
        { //temp foreach
          tuple2_L_L* e4 = agg1.head;
          while(e4){
            long bob = e4->_1;
            long v5 = e4->__av;
            COUNT.addOrDelOnZero(se1.modify(bob,r_b),v5);
            e4 = e4->nxt;
          }
        }
        COUNT_mR2.addOrDelOnZero(se2.modify(r_a,r_b),1L);
        COUNT_mR5.addOrDelOnZero(se3.modify(r_b,r_a),1L);
      }
      END_TRIGGER(exec_stats,"insert_R")
      END_TRIGGER(ivc_stats,"insert_R")
    }
    void unwrap_insert_R(const event_args_t& ea) {
      on_insert_R(*(reinterpret_cast<long*>(ea[0])), *(reinterpret_cast<long*>(ea[1])));
    }
    
    void on_delete_R(const long r_a, const long r_b) {
      BEGIN_TRIGGER(exec_stats,"delete_R")
      BEGIN_TRIGGER(ivc_stats,"delete_R")
      {  //++tN;
        add2.clear();
        { //foreach
          COUNT_mR2_entry* e5 = COUNT_mR2.head;
          while(e5){
            long r2_a = e5->R2_A;
            long joe = e5->JOE;
            long v7 = e5->__av;
            add_to_temp_map(add2, st4.modify(joe,((v7 * (r_b > r2_a)) * -1L)));
            e5 = e5->nxt;
          }
        }
        add_to_temp_map(add2, st5.modify(r_b,(r_b > r_a)));
        { //temp foreach
          tuple2_L_L* e6 = add2.head;
          while(e6){
            long joe = e6->_1;
            long v8 = e6->__av;
            COUNT.addOrDelOnZero(se4.modify(r_a,joe),v8);
            e6 = e6->nxt;
          }
        }
        agg2.clear();
        
        { //foreach
          COUNT_mR5_entry* e7 = COUNT_mR5.head;
          while(e7){
            long r1_b = e7->R1_B;
            long bob = e7->BOB;
            long v9 = e7->__av;
            add_to_temp_map(agg2, st6.modify(bob,(v9 * (r1_b > r_a))));
            e7 = e7->nxt;
          }
        }
        { //temp foreach
          tuple2_L_L* e8 = agg2.head;
          while(e8){
            long bob = e8->_1;
            long v10 = e8->__av;
            COUNT.addOrDelOnZero(se4.modify(bob,r_b),(v10 * -1L));
            e8 = e8->nxt;
          }
        }
        COUNT_mR2.addOrDelOnZero(se5.modify(r_a,r_b),-1L);
        COUNT_mR5.addOrDelOnZero(se6.modify(r_b,r_a),-1L);
      }
      END_TRIGGER(exec_stats,"delete_R")
      END_TRIGGER(ivc_stats,"delete_R")
    }
    void unwrap_delete_R(const event_args_t& ea) {
      on_delete_R(*(reinterpret_cast<long*>(ea[0])), *(reinterpret_cast<long*>(ea[1])));
    }
    
    void on_system_ready_event() {
      BEGIN_TRIGGER(exec_stats,"system_ready_event")
      BEGIN_TRIGGER(ivc_stats,"system_ready_event")
      {  //
        
      }
      END_TRIGGER(exec_stats,"system_ready_event")
      END_TRIGGER(ivc_stats,"system_ready_event")
    }
  
  private:
  
    /* Sample entries for avoiding recreation of temporary objects */
    COUNT_entry se1;
    tuple2_L_L st1;
    tuple2_L_L st2;
    tuple2_L_L st3;
    COUNT_mR2_entry se2;
    COUNT_mR5_entry se3;
    COUNT_entry se4;
    tuple2_L_L st4;
    tuple2_L_L st5;
    tuple2_L_L st6;
    COUNT_mR2_entry se5;
    COUNT_mR5_entry se6;
  
    /* Data structures used for storing materialized views */
    COUNT_mR2_map COUNT_mR2;
    COUNT_mR5_map COUNT_mR5;
    MultiHashMap<tuple2_L_L,long,HashIndex<tuple2_L_L,long> > agg2;
    MultiHashMap<tuple2_L_L,long,HashIndex<tuple2_L_L,long> > add2;
    MultiHashMap<tuple2_L_L,long,HashIndex<tuple2_L_L,long> > agg1;
    MultiHashMap<tuple2_L_L,long,HashIndex<tuple2_L_L,long> > add1;
    
  
  };
